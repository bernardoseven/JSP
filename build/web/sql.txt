create table cliente (
	rut varchar2(15),
	nombre varchar2(30),
	comuna varchar2(15),
	email varchar2(40),
	password varchar2(50),
	fono number(9,0),
	tipo varchar2(5));

create table cita (
	fecha date,
    	hora varchar2(5),
    	empleado varchar2(30),
	requerimiento varchar2(30),
	cliente varchar2(15),
	estado varchar2(15));

create table requerimiento (
